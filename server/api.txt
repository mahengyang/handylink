// get handylink from server.
url: angryang.com/cgi/ruby/user_info.rb

param：
   handle:"get" -- 固定
   username     -- 非必须
   mac          -- 非必须

return:
   [{url:"",title:""}, ...]

   url:网址
   title:网站标题
eg:
   http://angryang.com/cgi/ruby/user_info.rb?handle=get
   
// add new handylink.
url:
  angryang.com/cgi/ruby/user_info.rb

param:
  handle:"add" -- 固定
  url:"" -- 必须
  title:"" -- 必须
  mac:"" -- 非必须
  username: "" -- 非必须
  browser_version:"" --非必须

  http://angryang.com/cgi/ruby/user_info.rb?handle=add&url=www.taobao.com&title=%E6%B7%98%E5%AE%9D&mac=default
